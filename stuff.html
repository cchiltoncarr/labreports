import time
import matplotlib.pyplot as plt


startx = -2.0
endx = 1.0
starty = -1.5
endy = 1.5
n_points_per_axis = 100

maxiter = 30


def Mandelbrot_iter(x, y):
    it = 0
    z = complex(0, 0)
    c = complex(x, y)
    while abs(z) < 2 and it < maxiter:
        z = z**2 + c
        it += 1
    return it


xstep = (endx - startx) / (n_points_per_axis - 1)
ystep = (endy - starty) / (n_points_per_axis - 1)

x_values = []
y_values = []
color_values = []

for i in range(n_points_per_axis):
    x = startx + i * xstep
    print(f"Plotting for x = {x:.03f}")
    for j in range(n_points_per_axis):
        y = starty + j * ystep
        it = Mandelbrot_iter(x, y)
        col = it / maxiter
        x_values.append(x)
        y_values.append(y)
        color_values.append(col)
        time.sleep(0)

plt.scatter(x_values, y_values, c=color_values, s=9, cmap='plasma') # added color map plasma
plt.xlabel("X")
plt.ylabel("Y")
plt.title("Fractal")
plt.colorbar(label="Iterations/maxiter")
plt.show()
