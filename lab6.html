import numpy as np
import matplotlib.pyplot as plt

def generate_grid(size):
    x = np.linspace(1, 10, size)  
    y = np.linspace(1, 10, size)
    grid = np.meshgrid(x, y)  
    return grid

def plot_grid(grid):
    x_grid, y_grid = grid
    plt.figure(figsize=(5, 5))
    plt.scatter(x_grid, y_grid, s=2, color='purple') 
    plt.title(f'2D Grid of {len(x_grid)}x{len(y_grid)} points')
    plt.show()

size = 50  

grid = generate_grid(size)
plot_grid(grid)
