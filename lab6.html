import numpy as np
import matplotlib.pyplot as plt

def generate_grid(size):
    x = np.linspace(1, 10, size)  
    y = np.linspace(1, 10, size)
    grid = np.meshgrid(x, y)  
    return grid

def plot_grid(grid):
    x_grid, y_grid = grid
    plt.figure(figsize=(5, 5))
    plt.scatter(x_grid, y_grid, s=2, color='purple') 
    plt.title(f'2D Grid of {len(x_grid)}x{len(y_grid)} points')
    plt.show()

size = 50  

grid = generate_grid(size)
plot_grid(grid)

import numpy as np
import matplotlib.pyplot as plt
from mpl_toolkits.mplot3d import Axes3D

def generate_grid(size):
    x = np.linspace(-1, 1, size)
    y = np.linspace(-1, 1, size)
    z = np.linspace(-1, 1, size)
    x_grid, y_grid, z_grid = np.meshgrid(x, y, z)
    return x_grid, y_grid, z_grid

def plot_grid(x_grid, y_grid, z_grid):
    fig = plt.figure(figsize=(5, 5))
    ax = fig.add_subplot(111, projection='3d')

    ax.scatter(x_grid, y_grid, z_grid, s=2, color='purple')
    ax.set_title(f'3D grid of {x_grid.shape[0]}x{y_grid.shape[1]}x{z_grid.shape[2]} points')
    
    ax.set_xlabel('X')
    ax.set_ylabel('Y')
    ax.set_zlabel('Z')

    plt.show()

size = 30

x_grid, y_grid, z_grid = generate_grid(size)
plot_grid(x_grid, y_grid, z_grid)

Conclusion

It's very interesting to see the relationship between the 2D function-images and the same ones but in 3D instead. Since the first two I tried (z**2 and z**3) turned out to be cone-shaped, I am curious if there are other options besides that.

Seems to be that exponents (greater than 1) cause a cone shape, and roots (or exponents between 0 and 1) make more of a curved plane. Negative exponents seem to also create a cone, although the points are much further apart so the overall shape of the cone is not easily visualized.
